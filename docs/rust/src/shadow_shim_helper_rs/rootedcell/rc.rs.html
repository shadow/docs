14struct RootedRcInternal<T> {
15    val: UnsafeCell<Option<T>>,
16    strong_count: Cell<u32>,
17    weak_count: Cell<u32>,
18}
19
20impl<T> RootedRcInternal<T> {
21    pub fn new(val: T) -> Self {
22        Self {
23            val: UnsafeCell::new(Some(val)),
24            strong_count: Cell::new(1),
25            weak_count: Cell::new(0),
26        }
27    }
28
29    pub fn inc_strong(&self) {
30        self.strong_count.set(self.strong_count.get() + 1)
31    }
32
33    pub fn dec_strong(&self) {
34        self.strong_count.set(self.strong_count.get() - 1)
35    }
36
37    pub fn inc_weak(&self) {
38        self.weak_count.set(self.weak_count.get() + 1)
39    }
40
41    pub fn dec_weak(&self) {
42        self.weak_count.set(self.weak_count.get() - 1)
43    }
44}
45
46enum RefType {
47    Weak,
48    Strong,
49}
50
51// Shared implementation for strong and weak references
52struct RootedRcCommon<T> {
53    tag: Tag,
54    internal: Option<NonNull<RootedRcInternal<T>>>,
55}
56
57impl<T> RootedRcCommon<T> {
58    pub fn new(root: &Root, val: T) -> Self {
59        Self {
60            tag: root.tag(),
61            internal: Some(
62                NonNull::new(Box::into_raw(Box::new(RootedRcInternal::new(val)))).unwrap(),
63            ),
64        }
65    }
66
67    // Validates that no other thread currently has access to self.internal, and
68    // return a reference to it.
69    pub fn borrow_internal(&self, root: &Root) -> &RootedRcInternal<T> {
70        assert_eq!(
71            root.tag, self.tag,
72            "Tried using root {:?} instead of {:?}",
73            root.tag, self.tag
74        );
75        // SAFETY:
76        // * Holding a reference to `root` proves no other threads can currently
77        //   access `self.internal`.
78        // * `self.internal` is accessible since we hold a strong reference.
79        unsafe { self.internal.unwrap().as_ref() }
80    }
81
82    /// Decrement the reference. If this was the last strong reference, return the value.
83    pub fn safely_drop(mut self, root: &Root, t: RefType) -> Option<T> {
84        let internal: &RootedRcInternal<T> = self.borrow_internal(root);
85        match t {
86            RefType::Weak => internal.dec_weak(),
87            RefType::Strong => internal.dec_strong(),
88        };
89        let strong_count = internal.strong_count.get();
90        let weak_count = internal.weak_count.get();
91
92        // If there are no more strong references, prepare to drop the value.
93        // If the value was already dropped (e.g. because we're now dropping a
94        // weak reference after all the strong refs were already dropped), this
95        // is a no-op.
96        let val: Option<T> = if strong_count == 0 {
97            // SAFETY: Since no strong references remain, nothing else can be
98            // referencing the internal value.
99            unsafe { internal.val.get().as_mut().unwrap().take() }
100        } else {
101            None
102        };
103
104        // Clear `self.internal`, so that `drop` knows that `safely_drop` ran.
105        let internal: NonNull<RootedRcInternal<T>> = self.internal.take().unwrap();
106
107        // If there are neither strong nor weak references, drop `internal` itself.
108        if strong_count == 0 && weak_count == 0 {
109            // SAFETY: We know the pointer is still valid since we had the last
110            // reference, and since the counts are now zero, there can be no
111            // other references.
112            drop(unsafe { Box::from_raw(internal.as_ptr()) });
113        }
114
115        val
116    }
117
118    pub fn clone(&self, root: &Root, t: RefType) -> Self {
119        let internal: &RootedRcInternal<T> = self.borrow_internal(root);
120        match t {
121            RefType::Weak => internal.inc_weak(),
122            RefType::Strong => internal.inc_strong(),
123        };
124        Self {
125            tag: self.tag,
126            internal: self.internal,
127        }
128    }
129}

140impl<T> Drop for RootedRcCommon<T> {
141    #[inline]
142    fn drop(&mut self) {
143        if self.internal.is_some() {
144            // `explicit_drop` is the public interface for the internal `safely_drop`
145            log::error!("Dropped without calling `explicit_drop`");
146
147            // We *can* continue without violating Rust safety properties; the
148            // underlying object will just be leaked, since the ref count will
149            // never reach zero.
150            //
151            // If we're not already panicking, it's useful to panic here to make
152            // the leak more visible.
153            //
154            // If we are already panicking though, that may already explain how
155            // a call to `safely_drop` got skipped, and panicking again would
156            // just obscure the original panic.
157            #[cfg(debug_assertions)]
158            if !already_panicking() {
159                panic!("Dropped without calling `explicit_drop`");
160            }
161        }
162    }
163}

165// SAFETY: RootedRcCommon ensures that its internals can only be accessed when
166// the Root is held by the current thread, effectively synchronizing the
167// reference count.
168unsafe impl<T: Sync + Send> Send for RootedRcCommon<T> {}
169unsafe impl<T: Sync + Send> Sync for RootedRcCommon<T> {}
170
171/// Analagous to [std::rc::Rc]. In particular like [std::rc::Rc] and unlike
172/// [std::sync::Arc], it doesn't perform any atomic operations internally,
173/// making it relatively inexpensive
174///
175/// Unlike [std::rc::Rc], this type [Send] and [Sync] if `T` is. This is safe because
176/// the owner is required to prove ownership of the associated [Root]
177/// to perform any sensitive operations.
178///
179/// Instances must be destroyed explicitly, using [`RootedRc::explicit_drop`],
180/// [`RootedRc::explicit_drop_recursive`], or [`RootedRc::into_inner`].  These
181/// validate that the [Root] is held before manipulating reference counts, etc.
182///
183/// Dropping `RootedRc` without calling one of these methods results in a
184/// `panic` in debug builds, or leaking the object in release builds.
185pub struct RootedRc<T> {
186    common: RootedRcCommon<T>,
187}
188
189impl<T> RootedRc<T> {
190    /// Creates a new object associated with `root`.
191    #[inline]
192    pub fn new(root: &Root, val: T) -> Self {
193        Self {
194            common: RootedRcCommon::new(root, val),
195        }
196    }
197
198    /// Create a weak reference.
199    ///
200    /// We use fully qualified syntax here for consistency with Rc and Arc and
201    /// to avoid name conflicts with `T`'s methods.
202    #[inline]
203    pub fn downgrade(this: &Self, root: &Root) -> RootedRcWeak<T> {
204        RootedRcWeak {
205            common: this.common.clone(root, RefType::Weak),
206        }
207    }
208
209    /// Like [Clone::clone], but requires that the corresponding Root is held.
210    ///
211    /// Intentionally named clone to shadow Self::deref()::clone().
212    ///
213    /// Panics if `root` is not the associated [Root].
214    #[inline]
215    pub fn clone(&self, root: &Root) -> Self {
216        Self {
217            common: self.common.clone(root, RefType::Strong),
218        }
219    }
220
221    /// Drop the `RootedRc`, and return the inner value if this was the last
222    /// strong reference.
223    #[inline]
224    pub fn into_inner(this: Self, root: &Root) -> Option<T> {
225        this.common.safely_drop(root, RefType::Strong)
226    }
227
228    /// Drops `self`, and if `self` was the last strong reference, call
229    /// `ExplicitDrop::explicit_drop` on the internal value.
230    pub fn explicit_drop_recursive(
231        self,
232        root: &Root,
233        param: &T::ExplicitDropParam,
234    ) -> Option<T::ExplicitDropResult>
235    where
236        T: ExplicitDrop,
237    {
238        Self::into_inner(self, root).map(|val| val.explicit_drop(param))
239    }
240}
241
242impl<T> ExplicitDrop for RootedRc<T> {
243    type ExplicitDropParam = Root;
244
245    type ExplicitDropResult = ();
246
247    /// If T itself implements `ExplicitDrop`, consider
248    /// `RootedRc::explicit_drop_recursive` instead to call it when dropping the
249    /// last strong reference.
250    fn explicit_drop(self, root: &Self::ExplicitDropParam) -> Self::ExplicitDropResult {
251        self.common.safely_drop(root, RefType::Strong);
252    }
253}
254
255impl<T> core::ops::Deref for RootedRc<T> {
256    type Target = T;
257
258    #[inline]
259    fn deref(&self) -> &Self::Target {
260        // No need to require a reference to `Root` here since we're not
261        // touching the counts, only the value itself, which we already required
262        // to be Sync and Send for RootedRc<T> to be Sync and Send.
263
264        // SAFETY: Pointer to `internal` is valid, since we hold a strong ref.
265        let internal = unsafe { self.common.internal.unwrap().as_ref() };
266
267        // SAFETY: Since we hold a strong ref, we know that `val` is valid, and
268        // that there are no mutable references to it. (The only time we create
269        // a mutable reference is to drop the T value when the strong ref count
270        // reaches zero)
271        let val = unsafe { &*internal.val.get() };
272        val.as_ref().unwrap()
273    }
274}

433pub struct RootedRcWeak<T> {
434    common: RootedRcCommon<T>,
435}
436
437impl<T> RootedRcWeak<T> {
438    #[inline]
439    pub fn upgrade(&self, root: &Root) -> Option<RootedRc<T>> {
440        let internal = self.common.borrow_internal(root);
441
442        if internal.strong_count.get() == 0 {
443            return None;
444        }
445
446        Some(RootedRc {
447            common: self.common.clone(root, RefType::Strong),
448        })
449    }
450
451    /// Like [Clone::clone], but requires that the corresponding Root is held.
452    ///
453    /// Intentionally named clone to shadow Self::deref()::clone().
454    ///
455    /// Panics if `root` is not the associated [Root].
456    #[inline]
457    pub fn clone(&self, root: &Root) -> Self {
458        Self {
459            common: self.common.clone(root, RefType::Weak),
460        }
461    }
462}

464impl<T> ExplicitDrop for RootedRcWeak<T> {
465    type ExplicitDropParam = Root;
466
467    type ExplicitDropResult = ();
468
469    #[inline]
470    fn explicit_drop(self, root: &Self::ExplicitDropParam) -> Self::ExplicitDropResult {
471        let val = self.common.safely_drop(root, RefType::Weak);
472        // Since this isn't a strong reference, this can't be the *last* strong
473        // reference, so the value should never be returned.
474        debug_assert!(val.is_none());
475    }
476}

528    fn circular_reference() {
529        std::thread_local! {
530            static THREAD_ROOT: Root = Root::new();
531        }
532
533        struct MyStruct {
534            // Circular reference
535            weak_self: Cell<Option<RootedRcWeak<Self>>>,
536        }
537        impl MyStruct {
538            fn new() -> RootedRc<Self> {
539                THREAD_ROOT.with(|root| {
540                    let rv = RootedRc::new(
541                        root,
542                        MyStruct {
543                            weak_self: Cell::new(None),
544                        },
545                    );
546                    let weak = RootedRc::downgrade(&rv, root);
547                    rv.weak_self.set(Some(weak));
548                    rv
549                })
550            }
551        }
552        impl Drop for MyStruct {
553            fn drop(&mut self) {
554                let weak = self.weak_self.replace(None).unwrap();
555                THREAD_ROOT.with(|root| {
556                    weak.explicit_drop(root);
557                });
558            }
559        }
560
561        let val = MyStruct::new();
562        THREAD_ROOT.with(|root| {
563            val.explicit_drop(root);
564        })
