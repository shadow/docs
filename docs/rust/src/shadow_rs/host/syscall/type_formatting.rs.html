9fn fmt_int_with_suffix(
10    f: &mut std::fmt::Formatter<'_>,
11    ptr: u64,
12    suffix: &str,
13) -> std::fmt::Result {
14    if ptr == 0 {
15        write!(f, "{ptr:#x} <null>")
16    } else {
17        write!(f, "{ptr:#x} {suffix}")
18    }
19}

21fn fmt_ptr_with_suffix<T>(
22    f: &mut std::fmt::Formatter<'_>,
23    ptr: ForeignPtr<T>,
24    suffix: &str,
25) -> std::fmt::Result {
26    if ptr.is_null() {
27        write!(f, "{ptr:p} <null>")
28    } else {
29        write!(f, "{ptr:p} {suffix}")
30    }

33/// Implement `SyscallDisplay` using its `Display` implementation. The type must implement
34/// `TryFromSyscallReg`.
35macro_rules! simple_display_impl {
36    ($type:ty, $($types:ty),+) => {
37        simple_display_impl!($type);
38        simple_display_impl!($($types),+);
39    };
40    ($type:ty) => {
41        impl SyscallDisplay for SyscallVal<'_, $type> {
42            fn fmt(
43                &self,
44                f: &mut std::fmt::Formatter<'_>,
45                _options: FmtOptions,
46                _mem: &MemoryManager,
47            ) -> std::fmt::Result {
48                match <$type>::try_from(self.reg).ok() {
49                    Some(x) => write!(f, "{x}"),
50                    // if the conversion to type T was unsuccessful, just show an integer
51                    None => fmt_int_with_suffix(f, u64::from(self.reg), "<invalid-value>"),
52                }
53            }
54        }
55    };
56}

83/// Implement `SyscallDisplay` for a bitflags type. The type must implement `TryFromSyscallReg`.
84macro_rules! bitflags_impl {
85    ($type:ty, $($types:ty),+) => {
86        bitflags_impl!($type);
87        bitflags_impl!($($types),+);
88    };
89    ($type:ty) => {
90        impl SyscallDisplay for SyscallVal<'_, $type> {
91            fn fmt(
92                &self,
93                f: &mut std::fmt::Formatter<'_>,
94                _options: FmtOptions,
95                _mem: &MemoryManager,
96            ) -> std::fmt::Result {
97                match <$type>::try_from(self.reg).ok() {
98                    Some(x) => {
99                        if x.is_empty() {
100                            write!(f, "(empty)")
101                        } else {
102                            bitflags::parser::to_writer(&x, f)
103                        }
104                    }
105                    // if the conversion to type T was unsuccessful, just show an integer
106                    None => fmt_int_with_suffix(f, u64::from(self.reg), "<invalid-value>"),
107                }
108            }
109        }
110    };

113/// Display the pointer and data. Accesses plugin memory. Can only be used for pod types (enforced
114/// by the memory manager).
115macro_rules! deref_pointer_impl {
116    ($type:ty, $($types:ty),+) => {
117        deref_pointer_impl!($type);
118        deref_pointer_impl!($($types),+);
119    };
120    ($type:ty) => {
121        impl SyscallDisplay for SyscallVal<'_, *const $type> {
122            fn fmt(
123                &self,
124                f: &mut std::fmt::Formatter<'_>,
125                options: FmtOptions,
126                mem: &MemoryManager,
127            ) -> std::fmt::Result {
128                let ptr = ForeignPtr::<$type>::from(self.reg);
129                match (options, mem.memory_ref(ForeignArrayPtr::new(ptr, 1))) {
130                    (FmtOptions::Standard, Ok(vals)) => write!(f, "{:?} ({:p})", &(*vals)[0], ptr),
131                    // if we couldn't read the memory, just show the pointer instead
132                    (FmtOptions::Standard, Err(_)) => fmt_ptr_with_suffix(f, ptr, "<invalid-read>"),
133                    (FmtOptions::Deterministic, _) => write!(f, "<pointer>"),
134                }
135            }
136        }
137    };
138}

236fn fmt_buffer(
237    f: &mut std::fmt::Formatter<'_>,
238    ptr: ForeignPtr<u8>,
239    len: usize,
240    options: FmtOptions,
241    mem: &MemoryManager,
242) -> std::fmt::Result {
243    const DISPLAY_LEN: usize = 40;
244
245    if options == FmtOptions::Deterministic {
246        return write!(f, "<pointer>");
247    }
248
249    let mem_ref = match mem.memory_ref_prefix(ForeignArrayPtr::new(ptr, len)) {
250        Ok(x) => x,
251        // the pointer didn't reference any valid memory
252        Err(_) => return fmt_ptr_with_suffix(f, ptr, "<invalid-addr>"),
253    };
254
255    let mut s = String::with_capacity(DISPLAY_LEN);
256
257    // the number of plugin mem bytes used; num_bytes <= s.len()
258    let mut num_plugin_bytes = 0;
259
260    for c in mem_ref.iter() {
261        let escaped = std::ascii::escape_default(*c);
262
263        if s.len() + escaped.len() > DISPLAY_LEN {
264            break;
265        }
266
267        for b in escaped {
268            s.push(b.into())
269        }
270
271        num_plugin_bytes += 1;
272    }
273
274    if len > num_plugin_bytes {
275        write!(f, "\"{s}\"...")
276    } else {
277        write!(f, "\"{s}\"")
278    }
279}

281fn fmt_string(
282    f: &mut std::fmt::Formatter<'_>,
283    ptr: ForeignPtr<u8>,
284    len: Option<usize>,
285    mem: &MemoryManager,
286) -> std::fmt::Result {
287    const DISPLAY_LEN: usize = 40;
288
289    // the pointer may point to a buffer of unknown length, so we may have to choose our own size
290    let len = len.unwrap_or(
291        // read up to one extra character to check if it's a NUL byte
292        //
293        // each byte may take 1 byte to display (ex: 0x41 -> "A") or up to 4 bytes to display (ex:
294        // 0x00 -> "\x00"), so a buffer of size `DISPLAY_LEN + 1` should always be enough space to
295        // print a string of length `DISPLAY_LEN`
296        DISPLAY_LEN + 1,
297    );
298
299    let mem_ref = match mem.memory_ref_prefix(ForeignArrayPtr::new(ptr, len)) {
300        Ok(x) => x,
301        // the pointer didn't reference any valid memory
302        Err(_) => return fmt_ptr_with_suffix(f, ptr, "<invalid-addr>"),
303    };
304
305    let mut s = String::with_capacity(DISPLAY_LEN);
306
307    // the number of plugin mem bytes used; num_bytes <= s.len()
308    let mut found_nul = false;
309
310    for c in mem_ref.iter() {
311        // if it's a NUL byte, it's the end of the string
312        if *c == 0 {
313            found_nul = true;
314            break;
315        }
316
317        let escaped = std::ascii::escape_default(*c);
318
319        if s.len() + escaped.len() > DISPLAY_LEN {
320            break;
321        }
322
323        for b in escaped {
324            s.push(b.into())
325        }
326    }
327
328    if found_nul {
329        write!(f, "\"{s}\"")
330    } else {
331        write!(f, "\"{s}\"...")
332    }
333}

335/// Format a plugin's `libc::msghdr`. Any pointers contained in the `libc::msghdr` must be pointers
336/// within the plugin's memory space.
337fn fmt_msghdr(
338    f: &mut std::fmt::Formatter<'_>,
339    msg: &libc::msghdr,
340    _options: FmtOptions,
341    mem: &MemoryManager,
342) -> std::fmt::Result {
343    let ptr = ForeignPtr::from_raw_ptr(msg.msg_name as *mut u8);
344
345    // read the socket address from `msg.msg_name`
346    let addr = read_sockaddr(mem, ptr, msg.msg_namelen);
347
348    // prepare the socket address for formatting
349    let msg_name = DebugFormatter(move |fmt| {
350        match addr {
351            Ok(Some(addr)) => write!(fmt, "{addr} ({ptr:p})"),
352            Ok(None) => write!(fmt, "{ptr:p}"),
353            // if we weren't able to read the sockaddr (NULL, EFAULT, etc), just show the pointer
354            Err(_) => fmt_ptr_with_suffix(fmt, ptr, "<invalid-read>"),
355        }
356    });
357
358    // prepare the message flags for formatting
359    let msg_flags =
360        DebugFormatter(
361            move |fmt| match nix::sys::socket::MsgFlags::from_bits(msg.msg_flags) {
362                Some(x) => write!(fmt, "{x:?}"),
363                None => write!(fmt, "{:#x} <invalid>", msg.msg_flags),
364            },
365        );
366
367    // format msg
368    f.debug_struct("msghdr")
369        .field("msg_name", &msg_name)
370        .field("msg_namelen", &msg.msg_namelen)
371        .field("msg_iov", &msg.msg_iov)
372        .field("msg_iovlen", &msg.msg_iovlen)
373        .field("msg_control", &msg.msg_control)
374        .field("msg_controllen", &msg.msg_controllen)
375        .field("msg_flags", &msg_flags)
376        .finish()?;
377
378    Ok(())
379}

412pub struct SyscallSockAddrArg<const LEN_INDEX: usize> {}
413
414impl<const LEN_INDEX: usize> SyscallDisplay for SyscallVal<'_, SyscallSockAddrArg<LEN_INDEX>> {
415    fn fmt(
416        &self,
417        f: &mut std::fmt::Formatter<'_>,
418        options: FmtOptions,
419        mem: &MemoryManager,
420    ) -> std::fmt::Result {
421        if options == FmtOptions::Deterministic {
422            return write!(f, "<pointer>");
423        }
424
425        let ptr = self.reg.into();
426        let len = self.args[LEN_INDEX].into();
427
428        let Ok(addr) = read_sockaddr(mem, ptr, len) else {
429            return fmt_ptr_with_suffix(f, ptr, "<invalid-read>");
430        };
431
432        let Some(addr) = addr else {
433            return write!(f, "{ptr:p}");
434        };
435
436        write!(f, "{addr}")
437    }
438}

440/// A wrapper type to indicate that the value is non-deterministic.
441///
442/// This is for types that are typically deterministic (for example `u64`), but aren't deterministic
443/// for a particular syscall. For example, the last argument to futex() is a `u32` but may be
444/// optional depending on the futex operation, so the value given by the application may have been
445/// uninitialized, which Shadow sees as some arbitrary value.
446pub struct SyscallNonDeterministicArg<T> {
447    _phantom: std::marker::PhantomData<T>,
448}
449
450impl<'a, T> SyscallDisplay for SyscallVal<'a, SyscallNonDeterministicArg<T>>
451where
452    SyscallVal<'a, T>: SyscallDisplay,
453{
454    fn fmt(
455        &self,
456        f: &mut std::fmt::Formatter<'_>,
457        options: FmtOptions,
458        mem: &MemoryManager,
459    ) -> std::fmt::Result {
460        // if the user wants deterministic strace output, then we can't show the value
461        if options == FmtOptions::Deterministic {
462            return write!(f, "<non-deterministic>");
463        }
464
465        // otherwise format as per the inner type
466        self.cast::<T>().fmt(f, options, mem)
467    }
468}

470impl SyscallDisplay for SyscallVal<'_, *const libc::msghdr> {
471    fn fmt(
472        &self,
473        f: &mut std::fmt::Formatter<'_>,
474        options: FmtOptions,
475        mem: &MemoryManager,
476    ) -> std::fmt::Result {
477        let ptr: ForeignPtr<libc::msghdr> = self.reg.into();
478
479        if options == FmtOptions::Deterministic {
480            return write!(f, "<pointer>");
481        }
482
483        // read the msghdr
484        let ptr = ForeignArrayPtr::new(ptr, 1);
485        let Ok(msg) = mem.memory_ref(ptr) else {
486            // if we couldn't read the memory, just show the pointer instead
487            return fmt_ptr_with_suffix(f, ptr.ptr(), "<invalid-read>");
488        };
489        let msg = &(*msg)[0];
490
491        // format the msghdr
492        fmt_msghdr(f, msg, options, mem)?;
493
494        // show the original pointer
495        write!(f, " ({:p})", ptr.ptr())
496    }
