26impl<'this> RustcCodegenFlags<'this> {
27    // Parse flags obtained from CARGO_ENCODED_RUSTFLAGS
28    pub(crate) fn parse(rustflags_env: &'this str) -> Result<Self, Error> {
29        fn is_flag_prefix(flag: &str) -> bool {
30            [
31                "-Z",
32                "-C",
33                "--codegen",
34                "-L",
35                "-l",
36                "-o",
37                "-W",
38                "--warn",
39                "-A",
40                "--allow",
41                "-D",
42                "--deny",
43                "-F",
44                "--forbid",
45            ]
46            .contains(&flag)
47        }
48
49        fn handle_flag_prefix<'a>(prev: &'a str, curr: &'a str) -> (&'a str, &'a str) {
50            match prev {
51                "--codegen" | "-C" => ("-C", curr),
52                // Handle flags passed like --codegen=code-model=small
53                _ if curr.starts_with("--codegen=") => ("-C", &curr[10..]),
54                "-Z" => ("-Z", curr),
55                "-L" | "-l" | "-o" => (prev, curr),
56                // Handle lint flags
57                "-W" | "--warn" => ("-W", curr),
58                "-A" | "--allow" => ("-A", curr),
59                "-D" | "--deny" => ("-D", curr),
60                "-F" | "--forbid" => ("-F", curr),
61                _ => ("", curr),
62            }
63        }
64
65        let mut codegen_flags = Self::default();
66
67        let mut prev_prefix = None;
68        for curr in rustflags_env.split("\u{1f}") {
69            let prev = prev_prefix.take().unwrap_or("");
70            if prev.is_empty() && is_flag_prefix(curr) {
71                prev_prefix = Some(curr);
72                continue;
73            }
74
75            let (prefix, rustc_flag) = handle_flag_prefix(prev, curr);
76            codegen_flags.set_rustc_flag(prefix, rustc_flag)?;
77        }
78
79        Ok(codegen_flags)
80    }
81
82    fn set_rustc_flag(&mut self, prefix: &str, flag: &'this str) -> Result<(), Error> {
83        // Convert a textual representation of a bool-like rustc flag argument into an actual bool
84        fn arg_to_bool(arg: impl AsRef<str>) -> Option<bool> {
85            match arg.as_ref() {
86                "y" | "yes" | "on" | "true" => Some(true),
87                "n" | "no" | "off" | "false" => Some(false),
88                _ => None,
89            }
90        }
91
92        fn arg_to_u32(arg: impl AsRef<str>) -> Option<u32> {
93            arg.as_ref().parse().ok()
94        }
95
96        let (flag, value) = if let Some((flag, value)) = flag.split_once('=') {
97            (flag, Some(value))
98        } else {
99            (flag, None)
100        };
101        let flag = if prefix.is_empty() {
102            Cow::Borrowed(flag)
103        } else {
104            Cow::Owned(format!("{prefix}{flag}"))
105        };
106        let flag = flag.as_ref();
107
108        fn flag_not_empty_generic<T>(
109            flag: &str,
110            flag_value: Option<T>,
111        ) -> Result<Option<T>, Error> {
112            if let Some(flag_value) = flag_value {
113                Ok(Some(flag_value))
114            } else {
115                Err(Error::new(
116                    ErrorKind::InvalidFlag,
117                    format!("{flag} must have a value"),
118                ))
119            }
120        }
121        let flag_not_empty = |flag_value| flag_not_empty_generic(flag, flag_value);
122
123        match flag {
124            // https://doc.rust-lang.org/rustc/codegen-options/index.html#code-model
125            "-Ccode-model" => {
126                self.code_model = flag_not_empty(value)?;
127            }
128            // https://doc.rust-lang.org/rustc/codegen-options/index.html#no-vectorize-loops
129            "-Cno-vectorize-loops" => self.no_vectorize_loops = true,
130            // https://doc.rust-lang.org/rustc/codegen-options/index.html#no-vectorize-slp
131            "-Cno-vectorize-slp" => self.no_vectorize_slp = true,
132            // https://doc.rust-lang.org/rustc/codegen-options/index.html#profile-generate
133            "-Cprofile-generate" => {
134                self.profile_generate = flag_not_empty(value)?;
135            }
136            // https://doc.rust-lang.org/rustc/codegen-options/index.html#profile-use
137            "-Cprofile-use" => {
138                self.profile_use = flag_not_empty(value)?;
139            }
140            // https://doc.rust-lang.org/rustc/codegen-options/index.html#control-flow-guard
141            "-Ccontrol-flow-guard" => self.control_flow_guard = value.or(Some("true")),
142            // https://doc.rust-lang.org/rustc/codegen-options/index.html#lto
143            "-Clto" => self.lto = value.or(Some("true")),
144            // https://doc.rust-lang.org/rustc/linker-plugin-lto.html
145            "-Clinker-plugin-lto" => self.linker_plugin_lto = Some(true),
146            // https://doc.rust-lang.org/rustc/codegen-options/index.html#relocation-model
147            "-Crelocation-model" => {
148                self.relocation_model = flag_not_empty(value)?;
149            }
150            // https://doc.rust-lang.org/rustc/codegen-options/index.html#embed-bitcode
151            "-Cembed-bitcode" => self.embed_bitcode = value.map_or(Some(true), arg_to_bool),
152            // https://doc.rust-lang.org/rustc/codegen-options/index.html#force-frame-pointers
153            "-Cforce-frame-pointers" => {
154                self.force_frame_pointers = value.map_or(Some(true), arg_to_bool)
155            }
156            // https://doc.rust-lang.org/rustc/codegen-options/index.html#no-redzone
157            "-Cno-redzone" => self.no_redzone = value.map_or(Some(true), arg_to_bool),
158            // https://doc.rust-lang.org/rustc/codegen-options/index.html#soft-float
159            // Note: This flag is now deprecated in rustc.
160            "-Csoft-float" => self.soft_float = value.map_or(Some(true), arg_to_bool),
161            // https://doc.rust-lang.org/beta/unstable-book/compiler-flags/branch-protection.html
162            // FIXME: Drop the -Z variant and update the doc link once the option is stabilised
163            "-Zbranch-protection" | "-Cbranch-protection" => {
164                self.branch_protection = flag_not_empty(value)?;
165            }
166            // https://doc.rust-lang.org/beta/unstable-book/compiler-flags/dwarf-version.html
167            // FIXME: Drop the -Z variant and update the doc link once the option is stablized
168            "-Zdwarf-version" | "-Cdwarf-version" => {
169                self.dwarf_version = flag_not_empty_generic(flag, value.and_then(arg_to_u32))?;
170            }
171            // https://github.com/rust-lang/rust/issues/114903
172            // FIXME: Drop the -Z variant and update the doc link once the option is stabilized
173            "-Zstack-protector" | "-Cstack-protector" => {
174                self.stack_protector = flag_not_empty(value)?;
175            }
176            _ => {}
177        }
178        Ok(())
179    }

181    // Rust and clang/cc don't agree on what equivalent flags should look like.
182    pub(crate) fn cc_flags(&self, build: &Build, tool: &mut Tool, target: &TargetInfo<'_>) {
183        let family = tool.family;
184        // Push `flag` to `flags` if it is supported by the currently used CC
185        let mut push_if_supported = |flag: OsString| {
186            if build
187                .is_flag_supported_inner(&flag, tool, target)
188                .unwrap_or(false)
189            {
190                tool.args.push(flag);
191            } else {
192                build.cargo_output.print_warning(&format!(
193                    "Inherited flag {flag:?} is not supported by the currently used CC"
194                ));
195            }
196        };
197
198        let clang_or_gnu =
199            matches!(family, ToolFamily::Clang { .. }) || matches!(family, ToolFamily::Gnu);
200
201        // Flags shared between clang and gnu
202        if clang_or_gnu {
203            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-mbranch-protection
204            // https://gcc.gnu.org/onlinedocs/gcc/AArch64-Options.html#index-mbranch-protection (Aarch64)
205            // https://gcc.gnu.org/onlinedocs/gcc/ARM-Options.html#index-mbranch-protection-1 (ARM)
206            // https://developer.arm.com/documentation/101754/0619/armclang-Reference/armclang-Command-line-Options/-mbranch-protection
207            if let Some(value) = self.branch_protection {
208                push_if_supported(
209                    format!("-mbranch-protection={}", value.replace(",", "+")).into(),
210                );
211            }
212            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-mcmodel
213            // https://gcc.gnu.org/onlinedocs/gcc/Option-Summary.html (several archs, search for `-mcmodel=`).
214            // FIXME(madsmtm): Parse the model, to make sure we pass the correct value (depending on arch).
215            if let Some(value) = self.code_model {
216                push_if_supported(format!("-mcmodel={value}").into());
217            }
218            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-fno-vectorize
219            // https://gcc.gnu.org/onlinedocs/gnat_ugn/Vectorization-of-loops.html
220            if self.no_vectorize_loops {
221                push_if_supported("-fno-vectorize".into());
222            }
223            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-fno-slp-vectorize
224            // https://gcc.gnu.org/onlinedocs/gnat_ugn/Vectorization-of-loops.html
225            if self.no_vectorize_slp {
226                push_if_supported("-fno-slp-vectorize".into());
227            }
228            if let Some(value) = self.relocation_model {
229                let cc_flag = match value {
230                    // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-fPIC
231                    // https://gcc.gnu.org/onlinedocs/gcc/Code-Gen-Options.html#index-fPIC
232                    "pic" => Some("-fPIC"),
233                    // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-fPIE
234                    // https://gcc.gnu.org/onlinedocs/gcc/Code-Gen-Options.html#index-fPIE
235                    "pie" => Some("-fPIE"),
236                    // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-mdynamic-no-pic
237                    // https://gcc.gnu.org/onlinedocs/gcc/RS_002f6000-and-PowerPC-Options.html#index-mdynamic-no-pic
238                    "dynamic-no-pic" => Some("-mdynamic-no-pic"),
239                    _ => None,
240                };
241                if let Some(cc_flag) = cc_flag {
242                    push_if_supported(cc_flag.into());
243                }
244            }
245            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-fno-omit-frame-pointer
246            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-fomit-frame-pointer
247            // https://gcc.gnu.org/onlinedocs/gcc/Optimize-Options.html#index-fomit-frame-pointer
248            if let Some(value) = self.force_frame_pointers {
249                let cc_flag = if value {
250                    "-fno-omit-frame-pointer"
251                } else {
252                    "-fomit-frame-pointer"
253                };
254                push_if_supported(cc_flag.into());
255            }
256            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-mno-red-zone
257            // https://gcc.gnu.org/onlinedocs/gcc/x86-Options.html#index-mno-red-zone
258            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-mred-zone
259            // https://gcc.gnu.org/onlinedocs/gcc/x86-Options.html#index-mred-zone
260            if let Some(value) = self.no_redzone {
261                let cc_flag = if value { "-mno-red-zone" } else { "-mred-zone" };
262                push_if_supported(cc_flag.into());
263            }
264            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-msoft-float
265            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-mhard-float
266            // https://gcc.gnu.org/onlinedocs/gcc/Option-Summary.html (several archs, search for `-msoft-float`).
267            // https://gcc.gnu.org/onlinedocs/gcc/Option-Summary.html (several archs, search for `-mhard-float`).
268            if let Some(value) = self.soft_float {
269                let cc_flag = if value {
270                    "-msoft-float"
271                } else {
272                    // Do not use -mno-soft-float, that's basically just an alias for -mno-implicit-float.
273                    "-mhard-float"
274                };
275                push_if_supported(cc_flag.into());
276            }
277            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-gdwarf-2
278            // https://gcc.gnu.org/onlinedocs/gcc/Debugging-Options.html#index-gdwarf
279            if let Some(value) = self.dwarf_version {
280                push_if_supported(format!("-gdwarf-{value}").into());
281            }
282            // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-fstack-protector
283            // https://gcc.gnu.org/onlinedocs/gcc/Instrumentation-Options.html#index-fstack-protector
284            if let Some(value) = self.stack_protector {
285                // don't need to propagate stack-protector on MSVC since /GS is already the default
286                // https://learn.microsoft.com/en-us/cpp/build/reference/gs-buffer-security-check?view=msvc-170
287                //
288                // Do NOT `stack-protector=none` since it weakens security for C code,
289                // and `-Zstack-protector=basic` is deprecated and will be removed soon.
290                let cc_flag = match value {
291                    "strong" => Some("-fstack-protector-strong"),
292                    "all" => Some("-fstack-protector-all"),
293                    _ => None,
294                };
295                if let Some(cc_flag) = cc_flag {
296                    push_if_supported(cc_flag.into());
297                }
298            }
299        }
300
301        // Compiler-exclusive flags
302        match family {
303            ToolFamily::Clang { .. } => {
304                // GNU and Clang compilers both support the same PGO flags, but they use different libraries and
305                // different formats for the profile files which are not compatible.
306                // clang and rustc both internally use llvm, so we want to inherit the PGO flags only for clang.
307                // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-fprofile-generate
308                if let Some(value) = self.profile_generate {
309                    push_if_supported(format!("-fprofile-generate={value}").into());
310                }
311                // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-fprofile-use
312                if let Some(value) = self.profile_use {
313                    push_if_supported(format!("-fprofile-use={value}").into());
314                }
315
316                // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-fembed-bitcode
317                if let Some(value) = self.embed_bitcode {
318                    let cc_val = if value { "all" } else { "off" };
319                    push_if_supported(format!("-fembed-bitcode={cc_val}").into());
320                }
321
322                // https://doc.rust-lang.org/rustc/linker-plugin-lto.html
323                if self.linker_plugin_lto.unwrap_or(false) {
324                    // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-flto
325                    let cc_val = match self.lto {
326                        Some("thin") => "thin",
327                        _ => "full",
328                    };
329                    push_if_supported(format!("-flto={cc_val}").into());
330                }
331                // https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-mguard
332                if let Some(value) = self.control_flow_guard {
333                    let cc_val = match value {
334                        "y" | "yes" | "on" | "true" | "checks" => Some("cf"),
335                        "nochecks" => Some("cf-nochecks"),
336                        "n" | "no" | "off" | "false" => Some("none"),
337                        _ => None,
338                    };
339                    if let Some(cc_val) = cc_val {
340                        push_if_supported(format!("-mguard={cc_val}").into());
341                    }
342                }
343            }
344            ToolFamily::Gnu => {}
345            ToolFamily::Msvc { .. } => {
346                // https://learn.microsoft.com/en-us/cpp/build/reference/guard-enable-control-flow-guard
347                if let Some(value) = self.control_flow_guard {
348                    let cc_val = match value {
349                        "y" | "yes" | "on" | "true" | "checks" => Some("cf"),
350                        "n" | "no" | "off" | "false" => Some("cf-"),
351                        _ => None,
352                    };
353                    if let Some(cc_val) = cc_val {
354                        push_if_supported(format!("/guard:{cc_val}").into());
355                    }
356                }
357                // https://learn.microsoft.com/en-us/cpp/build/reference/oy-frame-pointer-omission
358                if let Some(value) = self.force_frame_pointers {
359                    // Flag is unsupported on 64-bit arches
360                    if !target.arch.contains("64") {
361                        let cc_flag = if value { "/Oy-" } else { "/Oy" };
362                        push_if_supported(cc_flag.into());
363                    }
364                }
365            }
366        }
367    }
